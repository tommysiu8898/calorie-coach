3. Backend Logic (Replit Express + DB)

Helper functions to get/update health flags:

```javascript
import Database from '@replit/database';
const db = new Database();

async function getHealthFlags(userId) {
  const res = await db.query('SELECT health_flags FROM users WHERE id = $1', [userId]);
  return res[0]?.health_flags || {};
}

async function updateHealthFlags(userId, updates) {
  const current = await getHealthFlags(userId);
  const updated = { ...current, ...updates };
  await db.query('UPDATE users SET health_flags = $1 WHERE id = $2', [updated, userId]);
}
```

Chat endpoint with missing flag detection:

```javascript
app.post('/api/chat/coach', async (req, res) => {
  const { userId, message } = req.body;
  let flags = await getHealthFlags(userId);
  
  // Define required flags for supplement advice
  const required = ['hypertension', 'high_cholesterol', 'diabetes', 'pregnant', 'medications', 'allergies'];
  const missing = required.filter(f => flags[f] === undefined || flags[f] === null);
  
  // Build prompt
  const prompt = SUPPLEMENT_SAFETY_PROMPT.replace('{{HEALTH_FLAGS}}', JSON.stringify(flags));
  
  // Call LLM (Kimi VL via OpenRouter)
  const aiReply = await callLLM(prompt, message);
  
  // Simple extraction: look for yes/no answers in user message and update flags
  const lowerMsg = message.toLowerCase();
  const updates = {};
  if (lowerMsg.includes('high blood pressure')) updates.hypertension = true;
  if (lowerMsg.includes('no high blood pressure') || lowerMsg.includes('don\'t have high blood pressure')) updates.hypertension = false;
  // similarly for cholesterol, diabetes, pregnant
  if (lowerMsg.includes('medication') || lowerMsg.includes('medicine')) {
    // can capture list, but for MVP just set true
    updates.medications = true;
  }
  if (lowerMsg.includes('allergic')) updates.allergies = true;
  
  if (Object.keys(updates).length) {
    await updateHealthFlags(userId, updates);
  }
  
  res.json({ reply: aiReply });
});
```
