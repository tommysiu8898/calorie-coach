// api/search-nearby.js - for Express.js on Replit
const axios = require('axios');

// Keyword mapping (Chinese → English) used only when app language = 'zh'
const keywordMap = {
  '健身房': 'gym',
  '瑜伽馆': 'yoga studio',
  '游泳池': 'swimming pool',
  '公园': 'park',
  '运动中心': 'sports center',
  '羽毛球馆': 'badminton court',
  '网球场': 'tennis court',
  '篮球场': 'basketball court',
  '跑道': 'running track'
};

// Helper: translate Chinese query to English keyword
function translateQuery(query, appLanguage) {
  if (appLanguage !== 'zh') return query;
  for (const [chinese, english] of Object.entries(keywordMap)) {
    if (query.includes(chinese)) return english;
  }
  return query; // fallback – keep original
}

// Main search function with timeout
async function searchNearbyPlaces(query, lat, lng, appLanguage, apiKey) {
  const translatedQuery = translateQuery(query, appLanguage);
  const searchTerm = `${translatedQuery} near me`;
  
  const url = 'https://maps.googleapis.com/maps/api/place/textsearch/json';
  const params = {
    query: searchTerm,
    location: `${lat},${lng}`,
    radius: 5000,        // 5km radius
    language: appLanguage === 'zh' ? 'zh-CN' : 'en',
    key: apiKey
  };

  // Timeout after 5 seconds
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), 5000);

  try {
    const response = await axios.get(url, { params, signal: controller.signal });
    clearTimeout(timeoutId);
    return response.data.results.map(place => ({
      name: place.name,
      address: place.formatted_address,
      rating: place.rating,
      lat: place.geometry.location.lat,
      lng: place.geometry.location.lng,
      types: place.types
    }));
  } catch (error) {
    clearTimeout(timeoutId);
    if (error.code === 'ECONNABORTED' || error.name === 'AbortError') {
      throw new Error('Request timeout – please try again');
    }
    throw new Error(`Places API error: ${error.message}`);
  }
}

// Express route handler
module.exports = async (req, res) => {
  // Expect JSON body: { query, lat, lng, appLanguage }
  const { query, lat, lng, appLanguage } = req.body;
  
  if (!query || !lat || !lng || !appLanguage) {
    return res.status(400).json({ error: 'Missing required fields: query, lat, lng, appLanguage' });
  }

  const apiKey = process.env.GOOGLE_PLACES_API_KEY;
  if (!apiKey) {
    return res.status(500).json({ error: 'Server misconfigured: missing Google API key' });
  }

  try {
    const results = await searchNearbyPlaces(query, lat, lng, appLanguage, apiKey);
    res.json({ success: true, results });
  } catch (err) {
    res.status(500).json({ success: false, error: err.message });
  }
};

// Use a static product list for MVP (replace with dynamic API later)
const iHerbAffiliateId = 'YOUR_AFFILIATE_ID'; // e.g., '1234567'

const productCatalog = [
  {
    id: 'prot123',
    name: 'Whey Protein Isolate',
    category: 'protein',
    iherbCode: 'PROT123',   // actual product code on iHerb
    description: 'Fast‑absorbing protein for muscle recovery.'
  },
  {
    id: 'omega456',
    name: 'Omega‑3 Fish Oil',
    category: 'omega',
    iherbCode: 'FISH456',
    description: 'High‑potency EPA/DHA for heart & brain health.'
  },
  {
    id: 'multi789',
    name: 'Daily Multivitamin',
    category: 'multivitamin',
    iherbCode: 'MULTI789',
    description: 'Essential vitamins and minerals for overall wellness.'
  }
];

function getAffiliateLink(productCode) {
  // iHerb affiliate link format:
  // https://www.iherb.com/pr/[product-code]?rcode=[affiliate-id]
  return `https://www.iherb.com/pr/${productCode}?rcode=${iHerbAffiliateId}`;
}

module.exports = async (req, res) => {
  const { goal, appLanguage } = req.body;
  // Simple keyword matching to recommend products
  let recommended = [];
  const lowerGoal = (goal || '').toLowerCase();
  
  if (lowerGoal.includes('protein') || lowerGoal.includes('muscle')) {
    recommended = productCatalog.filter(p => p.category === 'protein');
  } else if (lowerGoal.includes('omega') || lowerGoal.includes('fish oil') || lowerGoal.includes('heart')) {
    recommended = productCatalog.filter(p => p.category === 'omega');
  } else {
    recommended = productCatalog; // fallback
  }
  
  const results = recommended.map(p => ({
    name: p.name,
    description: p.description,
    affiliateLink: getAffiliateLink(p.iherbCode),
    // No fake discount – just standard iHerb price
    priceNote: 'Price as shown on iHerb'
  }));
  
  res.json({ success: true, results, message: appLanguage === 'zh' ? '通过iHerb购买正品' : 'Shop genuine supplements on iHerb' });
};
